const newProjects = projects.filter((p) => { const created = new Date(p.startDate); const now = new Date(); const diffDays = (now - created) / (1000 * 60 * 60 * 24); return diffDays <= 7; }).length; 
const activeProjects = projects.filter((p) => { const now = new Date(); return ( new Date(p.startDate) <= now && new Date(p.endDate) >= now ); }).length; 
const incompleteProjects = projects.filter( (p) => p.completedSurgeries < p.totalTarget ).length; 
const completedProjects = projects.filter( (p) => p.completedSurgeries >= p.totalTarget ).length;
